文件1
a1.sources = r1
a1.sinks = k1 k2
a1.channels = c1

# 声明 Sink 组 g1
a1.sinkgroups = g1
# 声明 Sink 组 g1 的具体的 Sink k1 和 k2
a1.sinkgroups.g1.sinks = k1 k2
# 声明 Sink 组 g1 的逻辑处理器为 failover 故障转移
a1.sinkgroups.g1.processor.type = failover
# 组内 Sink 的权重值，<sinkName> 必须是当前组关联的 Sink 之一。数值越大越被优先使用
a1.sinkgroups.g1.processor.priority.k1 = 5
a1.sinkgroups.g1.processor.priority.k2 = 10
# 发生异常的 Sink 的最大故障转移时间（毫秒），默认为 30000 毫秒
a1.sinkgroups.g1.processor.maxpenalty = 10000

# 声明自定义拦截器（给请求头添加 timestamp）
a1.sources.r1.interceptors = ts
a1.sources.r1.interceptors.ts.type = com.yjxxt.flume.interceptor.EventInterceptor$Builder

a1.sources.r1.type = TAILDIR
# 通过一个 JSON 格式的文件记录每个文件的 Inode 信息、绝对路径和最近一次读取的位置
a1.sources.r1.positionFile = /var/log/flume/logdata_taildir2avro.json
# 定义文件组（被监控的文件夹目录集合），这些文件夹下的文件都会被监控，多个用空格分隔
a1.sources.r1.filegroups = f1
# a1.sources.r1.filegroups = f1 f2 f3
# 声明文件组 f1 监控的文件
a1.sources.r1.filegroups.f1 = /var/yjx/data/logs/app/event-.*log
# a1.sources.r1.filegroups.f2 = /var/yjx/data/logs/wxapp/event-.*log
# a1.sources.r1.filegroups.f3 = /var/yjx/data/logs/web/event-.*log
# headers.<filegroupName>.<headerKey>：给某个文件组下的 Event 添加一个固定的键值对到 header 中，值就是 value
a1.sources.r1.headers.f1.logType = app
# a1.sources.r1.headers.f2.logType = wxapp
# a1.sources.r1.headers.f3.logType = web
# 该参数控制从同一个文件中连续读取的最大批次数量，默认为 Long.MAX_VALUE
# 如果 Flume 同时监听多个文件，并且其中某个文件的写入速度远快于其他文件，那么其他文件有可能几乎无法被读取，所以强烈建议设定此参数
a1.sources.r1.maxBatchCount = 1000
# 一次读取数据行和写入 Channel 的最大数量
a1.sources.r1.batchSize = 5000

a1.sinks.k1.type = avro
a1.sinks.k1.hostname = node02
a1.sinks.k1.port = 55555

a1.sinks.k2.type = avro
a1.sinks.k2.hostname = node03
a1.sinks.k2.port = 55555

a1.channels.c1.type = file
# 检查点指的是内存队列在某一稳定时刻的“快照”，且每隔一段时间（checkpointInterval）File Channel 会将内存队列持久化到磁盘文件，也就是我们配置的检查点目录下
# 为了保证内存队列“快照”的完整性，在将内存队列持久到磁盘文件时需要锁定内存队列，也就是说，在这个过程中 Source 不能将 Event 写入到 Channel 中，并且 Sink 也不能从 Channel 中拉取 Event
a1.channels.c1.checkpointDir = /var/yjx/flume/checkpoint
# backupCheckpointDir 是检查点目录的备份目录，因为检查点文件经常被读写，很容易导致文件损坏
# a1.channels.c1.backupCheckpointDir = xxx
# 逗号分隔的数据目录列表，用于存储数据文件（dataDirs 可以配置多个不同磁盘下的目录，这样可以提高 File Channel 的性能）
a1.channels.c1.dataDirs = /var/yjx/flume/data
# Channel 的总容量大小，注意这个容量不是越大越好，配置越大一旦 Flume 挂掉丢失的 Event 也就越多
a1.channels.c1.capacity = 1000000
# Source 和 Sink 从 Channel 每次事务传输的 Event 数量，该配置要小于总容量大小
a1.channels.c1.transactionCapacity = 10000

a1.sources.r1.channels = c1
a1.sinks.k1.channel = c1
a1.sinks.k2.channel = c1

a1.sources = r1

a1.sinks.k1.type = hdfs
# %{logType}：Event header 中 key 为 logType 的值
# 这个 logType 可以是任意的 key，只要 header 中有就能读取，比如 %{abc} 将读取 header 中 key 为 aabc 的值
a1.sinks.k1.hdfs.path = hdfs://hdfs-yjx/yjx/%{logType}/ods/ods_%{logType}_event_log/dt=%Y-%m-%d
# Flume 在 HDFS 文件夹下创建新文件的固定前缀，默认为 FlumeData
a1.sinks.k1.hdfs.filePrefix = event
# 是否启用时间戳舍入机制（采用最简单的舍入机制：向 0 舍入，如果为 true，则影响除 %t 之外的所有基于时间的转义符）
a1.sinks.k1.hdfs.round = true
# 多少时间单位创建一个新的文件夹
a1.sinks.k1.hdfs.roundValue = 24
# 重新定义时间单位，可选值：second、minute、hour
a1.sinks.k1.hdfs.roundUnit = hour
# 解析存储目录路径时候所使用的时区名，例如：America/Los_Angeles、Asia/Shanghai
a1.sinks.k1.hdfs.timeZone = Asia/Shanghai
# 是否使用本地时间戳（而不是使用 Event Header 中自带的时间戳）
# true 处理时间，使用本地时间戳；false 事件时间，使用 Event Header 中自带的时间戳，时间戳属性 Key 必须为 timestamp
a1.sinks.k1.hdfs.useLocalTimeStamp = false
# 当前文件写入 Event 达到该数量后触发滚动创建新文件（0 表示不根据 Event 数量来分割文件）
a1.sinks.k1.hdfs.rollCount = 0
# 多久（当前文件写入达到该时间）触发滚动创建新文件，单位：Second（0 表示不按照时间来分割文件）
# 注意是写入达到该时间，假如 Flume 启动后数据源一直没有生成日志数据，则不会创建任何文件
a1.sinks.k1.hdfs.rollInterval = 3600
# 多大（当前文件写入达到该大小）触发滚动创建新文件，单位：Byte
a1.sinks.k1.hdfs.rollSize = 134217728
# 当前被打开的临时文件在该参数指定的时间（秒）内，没有任何数据写入，则将该临时文件关闭并重命名为目标文件，默认为 0
a1.sinks.k1.hdfs.idleTimeout = 30
# 积攒多少个 Event 才 flush 到 HDFS 一次
# 如果使用内存型 Channel 该配置要小于等于 Channel 的 transactionCapacity 配置
a1.sinks.k1.hdfs.batchSize = 5000
# 设置文件类型，支持压缩，默认为 SequenceFile。目前支持：SequenceFile、DataStream、CompressedStream
# DataStream 不会压缩文件，不需要设置 hdfs.codeC
# CompressedStream 压缩文件，必须设置 hdfs.codeC 参数
a1.sinks.k1.hdfs.fileType = DataStream
# 压缩算法。可选值：gzip、bzip2、lzo、lzop、snappy
#a1.sinks.k1.hdfs.codeC = gzip

a1.channels.c1.type = file

a1.sinks.k1.hdfs.path = hdfs://hdfs-yjx/yjx_gzip/%{logType}/ods/ods_%{logType}_event_log/dt=%Y-%m-%d

# 是否启用时间戳舍入机制（采用最简单的舍入机制：向 0 舍入，如果为 true，则影响除 %t 之外的所有基于时间的转义符）
a1.sinks.k1.hdfs.round = true
# 多少时间单位创建一个新的文件夹
a1.sinks.k1.hdfs.roundValue = 24
# 重新定义时间单位，可选值：second、minute、hour
a1.sinks.k1.hdfs.roundUnit = hour

a1.sinks.k1.hdfs.timeZone = Asia/Shanghai
# 是否使用本地时间戳（而不是使用 Event Header 中自带的时间戳）

a1.sinks.k1.hdfs.useLocalTimeStamp = false
# 当前文件写入 Event 达到该数量后触发滚动创建新文件（0 表示不根据 Event 数量来分割文件）

# 当前被打开的临时文件在该参数指定的时间（秒）内，没有任何数据写入，则将该临时文件关闭并重命名为目标文件，默认为 0
a1.sinks.k1.hdfs.idleTimeout = 30

a1.sinks.k1.hdfs.fileType = CompressedStream

a1.sinks.k1.hdfs.codeC = gzip
