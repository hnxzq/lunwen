    deviceId   STRING COMMENT '设备',
    account    STRING COMMENT '账号',
    score      DOUBLE COMMENT '评分',
    last_login STRING COMMENT '最后一次登录时间'
)
COMMENT '设备账号绑定评分表'
PARTITIONED BY (dt STRING)
STORED AS PARQUET
TBLPROPERTIES ('parquet.compression' = 'SNAPPY')
LOCATION 'hdfs://hdfs-yjx/yjx/app/dwd/dwd_device_account_bind_score';

-- false 表示关闭 SparkSQL 的 Parquet 文件解析器，使用 Hive 的 Parquet 文件解析器
SET spark.sql.hive.convertMetastoreParquet=false;

-- 根据条件查询 ods.ods_app_event_log 表，并创建临时视图
WITH v_event_log AS (
    SELECT
        deviceId,
        IF(account = '', 'null', account) AS account,
        sessionId,
        ts
    FROM ods.ods_app_event_log
    WHERE dt = '2024-07-06' AND deviceId IS NOT NULL AND sessionId IS NOT NULL
),
-- 计算每一台“设备-账号”的会话分数
v_combine_session_cur AS (
    SELECT
        deviceId,
        account,
        COUNT(*) * 100 AS score,
        MAX(ts) AS last_login
    FROM v_event_log
    GROUP BY deviceId, account
),
-- 获取前一日的设备账号绑定评分表数据
v_combine_session_prev AS (
    SELECT * FROM dwd.dwd_device_account_bind_score WHERE dt='2024-07-05'
)
INSERT OVERWRITE TABLE dwd.dwd_device_account_bind_score PARTITION(dt='2024-07-06')
SELECT
NVL(cur.deviceId, prev.deviceId) AS deviceId,
NVL(cur.account, prev.account) AS account,
CASE
    -- T-1 日（昨日设备账号绑定评分表）出现的组合，T 日（当前设备账号绑定评分表）没有出现，做 Score 衰减
    WHEN prev.score IS NOT NULL AND cur.score IS NULL THEN prev.score * 0.6
    -- T-1 日（昨日设备账号绑定评分表）出现的组合，T 日（当前设备账号绑定评分表）也出现，做 Score 累加
    WHEN prev.score IS NOT NULL AND cur.score IS NOT NULL THEN prev.score + cur.score
    -- T-1 日（昨日设备账号绑定评分表）没出现的组合，T 日（当前设备账号绑定评分表）出现，取 T 日的 Score
    WHEN prev.score IS NULL AND cur.score IS NOT NULL THEN cur.score
END AS score,
NVL(cur.last_login, prev.last_login) last_login
FROM v_combine_session_cur cur
FULL JOIN v_combine_session_prev prev ON cur.deviceId = prev.deviceId
    AND IF(cur.account=null, '', cur.account) = IF(prev.account=null, '', prev.account);

WITH order_info AS (
    SELECT
        sn, -- 订单编号
        member_id -- 用户编号
    FROM ods.ods_yjx_order_inc
    WHERE dt = '2024-07-06'

order_item AS (
    SELECT
        id, -- 编号
        order_sn, -- 订单编号
        create_time, -- 创建时间
        flow_price, -- 实际金额
        goods_price, -- 销售金额
        sku_id, -- SKU ID
        num -- 销售量
    FROM ods.ods_yjx_order_item_inc
    WHERE dt = '2024-07-06'
),
coupon_info AS (
    SELECT
        id, -- 购物券编号
        benefit_amount AS benefit_amount -- 优惠金额
    FROM ods.ods_yjx_coupon_info_full
    WHERE dt = '2024-07-06'
),
coupon_use AS (
    SELECT
        coupon_id, -- 购物券编号
        order_sn -- 订单编号
    FROM ods.ods_yjx_coupon_use_inc
    WHERE dt = '2024-07-06'

INSERT INTO TABLE dwd.dwd_yjx_order_item_inc PARTITION (dt = '2024-07-06')
SELECT
    order_item.id,
    order_item.order_sn,
    order_info.member_id,
    order_item.sku_id,
    order_item.create_time,
    -- 实际金额 - 优惠金额 / 商品数量 = 优惠价格
    order_item.flow_price - NVL(coupon_info.benefit_amount, 0) / order_item.num AS benefit_price,
    order_item.goods_price,
    order_item.num AS sku_num,
    -- 优惠总金额
    NVL(coupon_info.benefit_amount, 0) AS coupon_amount,
    -- 实际金额 * 商品数量 = 原价总金额
    order_item.flow_price * order_item.num AS original_amount,
    -- 实际金额 * 商品数量 - 优惠金额 = 总金额
    order_item.flow_price * order_item.num - NVL(coupon_info.benefit_amount, 0) AS total_price
FROM order_item
LEFT JOIN order_info ON order_item.order_sn = order_info.sn
LEFT JOIN coupon_use ON order_item.order_sn = coupon_use.order_sn
LEFT JOIN coupon_info ON coupon_use.coupon_id = coupon_info.id;

-- 根据日期每天增量拉取计算
WITH format_table AS (
    -- 2. 将 fetchCoupon 标记为 1，其他数据标记为 0
    SELECT
        guid,
        eventId,
        IF(eventId = 'fetchCoupon', 1, 0) AS flag
    FROM (
        -- 1. 先将用户的事件序列，按照归因分析需求定义的条件进行过滤，留下满足条件的事件
        SELECT guid, eventId, ts
        FROM dwd.dwd_app_event_detail
        WHERE dt = '2024-07-06'
        AND eventId IN ('fetchCoupon', 'adShow', 'goodsView', 'addCart')
        ORDER BY guid, ts
    ) AS t
),
flag_table AS (
    -- 3. 开窗聚合并将 flag 列从 UNBOUNDED PRECEDING 加到 CURRENT ROW
    -- 4. 用第三步计算的结果减去 flag 得到的结果就是根据特定数据 fetchCoupon 分组的结果
    SELECT
        guid,
        eventId,
        flag,
        SUM(flag) OVER(PARTITION BY guid ORDER BY guid
                   ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) - flag AS re_flag
    FROM format_table
),
result AS (
    -- 4. 根据 re_flag 分组，并将结果收集进集合，收集时排除 fetchCoupon 事件
    SELECT
        guid,
        CONCAT_WS(',', COLLECT_LIST(IF(eventId='fetchCoupon', null, eventId))) AS event_list
    FROM flag_table
    GROUP BY guid, re_flag
)
INSERT INTO TABLE dws.dws_attribution_compute PARTITION (dt = '2024-07-06')
SELECT
    '优惠券获取归因分析' AS attr_model,
    '线性归因' AS strategy,
    guid,
    'fetchCoupon' AS dest_event,
    event AS attr_event,
    ROUND(100 / SIZE(SPLIT(event_list, ',')), 2) AS weight
FROM result
LATERAL VIEW EXPLODE(SPLIT(event_list, ',')) event_table AS event
WHERE event_list <> '';

-- 聚合下单漏斗模型数据
WITH tmp AS (
    SELECT
        guid,
        -- 将 ts 和 eventId 使用 _ 进行拼接，并放入 List 集合，然后对集合排序，并使用逗号对集合数据进行拼接
        CONCAT_WS(',', SORT_ARRAY(COLLECT_LIST(CONCAT_WS('_', ts, eventId)))) AS event_seq
    FROM dwd.dwd_app_event_detail
    -- 根据下单漏斗模型过滤出指定事件类型的数据 goodsView, addCart, submitOrder
    WHERE dt = '2024-07-06' AND eventId IN ('goodsView', 'addCart', 'submitOrder')
    GROUP BY guid
)
INSERT INTO TABLE dws.dws_funnel_compute_step PARTITION (dt = '2024-07-06')
SELECT guid, funnel_model, eventId, max_step
FROM (
    SELECT
        guid,
        '下单漏斗模型' AS funnel_model,
        CASE
            -- 注意 CASE WHEN 的顺序
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?(addCart).*?(submitOrder).*?', 3) = 'submitOrder' THEN 'submitOrder'
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?(addCart).*?', 2) = 'addCart' THEN 'addCart'
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?', 1) = 'goodsView' THEN 'goodsView'
            ELSE ''
            END AS eventId,
        CASE
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?(addCart).*?(submitOrder).*?', 3) = 'submitOrder' then 3
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?(addCart).*?', 2) = 'addCart' THEN 2
            WHEN REGEXP_EXTRACT(event_seq, '.*?(goodsView).*?', 1) = 'goodsView' THEN 1
            ELSE 0
            END AS max_step
    FROM tmp
) AS t
WHERE max_step > 0;

CREATE EXTERNAL TABLE IF NOT EXISTS ads.ads_funnel_conversion_rate (
    calc_dt      STRING COMMENT '统计日期',
    funnel_model STRING COMMENT '漏斗模型名称',
    step         INT COMMENT '步骤',
    step_cnt     BIGINT COMMENT '步骤人数',
    rate         DOUBLE COMMENT '转化率'
)
COMMENT '漏斗模型转化率表'
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','

LOCATION 'hdfs://hdfs-yjx/yjx/app/ads/ads_funnel_conversion_rate';

-- 根据日期每天增量拉取计算
WITH funnel AS (
    SELECT
        funnel_model,
        SUM(IF(max_step >= 1, 1, 0)) AS step1,
        SUM(IF(max_step >= 2, 1, 0)) AS step2,
        SUM(IF(max_step >= 3, 1, 0)) AS step3
    FROM dws.dws_funnel_compute_step
    WHERE dt = '2024-07-06'
    GROUP BY funnel_model
),
format_funnel AS (
    SELECT
        funnel_model,
        step,
        step_cnt
    FROM funnel
    LATERAL VIEW EXPLODE(MAP(1,step1, 2,step2, 3,step3)) exploded_table AS step, step_cnt
)
INSERT INTO TABLE ads.ads_funnel_conversion_rate
SELECT
    '2024-07-06' AS calc_dt,
    funnel_model,
    step,
    step_cnt,
    -- 方案一：每一步人数都和第一步人数进行计算
    ROUND(step_cnt / FIRST_VALUE(step_cnt) OVER (PARTITION BY funnel_model ORDER BY step), 2) AS rate
    -- 方案二：每一步人数和上一步人数进行计算
    -- LAG(COL, N, DEFAULT_VAL)：往前第 N 行数据，没有数据的话用 DEFAULT_VAL 代替
    -- ROUND(step_cnt / LAG(step_cnt, 1, step_cnt) OVER (PARTITION BY funnel_model ORDER BY step), 2) AS rate
FROM format_funnel;

WITH order_item AS (

        member_id,
        brand_name,
        category1_name,

    LEFT JOIN dim.dim_sku_full AS sku ON o.sku_id = sku.id

    WHERE DATEDIFF('2024-07-06', o.dt) <= 30 AND DATEDIFF('2024-07-06', sku.dt) <= 30

        member_id,
        brand_name AS common_brand
    FROM (
        SELECT
            member_id,
            brand_name,

                member_id,
                brand_name, -- 品牌

WITH order_zip AS (
    SELECT
        member_id,
        MIN(create_time) AS first_order_time, -- MIN(订单创建时间)
        MAX(create_time) AS last_order_time, -- MAX(订单创建时间)
        -- COUNT( IF(当前日期与订单创建日期的差值 <= 30, 1, NULL) )
        COUNT(IF(DATEDIFF('2024-07-06', create_time) <= 30, 1, NULL)) AS month1_order_cnt,
        -- SUM( IF(当前日期与订单创建日期的差值 <= 30，总价格flow_price - 优惠金额discount_price, NULL) )
        SUM(IF(DATEDIFF('2024-07-06', create_time) <= 30, flow_price - NVL(discount_price, 0), NULL)) AS month1_order_amt,
        -- COUNT( IF(当前日期与订单创建日期的差值 <= 60, 1, NULL) )
        COUNT(IF(DATEDIFF('2024-07-06', create_time) <= 60, 1, NULL)) AS month2_order_cnt,
        -- SUM( IF(当前日期与订单创建日期的差值 <= 60，总价格flow_price - 优惠金额discount_price, NULL) )
        SUM(IF(DATEDIFF('2024-07-06', create_time) <= 60, flow_price - NVL(discount_price, 0), NULL)) AS month2_order_amt,
        -- COUNT( IF(当前日期与订单创建日期的差值 <= 90, 1, NULL) )
        COUNT(IF(DATEDIFF('2024-07-06', create_time) <= 90, 1, NULL)) AS month3_order_cnt,
        -- SUM( IF(当前日期与订单创建日期的差值 <= 90，总价格flow_price - 优惠金额discount_price, NULL) )
        SUM(IF(DATEDIFF('2024-07-06', create_time) <= 90, flow_price - NVL(discount_price, 0), NULL)) AS month3_order_amt,
        MAX(flow_price - NVL(discount_price, 0)) AS max_order_amt, -- 最大订单金额
        MIN(flow_price - NVL(discount_price, 0)) AS min_order_amt, -- 最小订单金额
        COUNT(*) AS total_order_cnt, -- 累计消费次数
        SUM(flow_price - NVL(discount_price, 0)) AS total_order_amt, -- 累计消费金额
        SUM(NVL(discount_price, 0)) AS total_coupon_amt -- 累计使用代金券金额
    FROM dwd.dwd_yjx_order_zip
    -- 最近30日
    WHERE zip_start_dt >= DATE_SUB('2024-07-06', 29) AND zip_start_dt <= '2024-07-06'
    GROUP BY member_id
),
address AS (
    SELECT
        member_id,
        consignee_address_path AS common_address
    FROM (
        SELECT
            member_id,
            consignee_address_path,
            ROW_NUMBER() OVER(PARTITION BY member_id ORDER BY addr_cnt DESC) AS r
        FROM (
            SELECT
                member_id,
                consignee_address_path, -- 收获地址
                COUNT(*) AS addr_cnt
            FROM dwd.dwd_yjx_order_zip
            WHERE zip_start_dt >= DATE_SUB('2024-07-06', 29) AND zip_start_dt <= '2024-07-06'
            GROUP BY member_id, consignee_address_path
        ) AS a
    ) AS b
    WHERE r = 1 -- 取第一就是最常用的收获地址
),
payment AS (
    SELECT
        member_id,
        payment_method AS common_paytype
    FROM (
        SELECT
            member_id,
            payment_method,
            ROW_NUMBER() OVER(PARTITION BY member_id ORDER BY payment_cnt DESC) AS r
        FROM (
            SELECT
                member_id,
                payment_method, -- 支付方式
                COUNT(*) AS payment_cnt
            FROM dwd.dwd_yjx_order_zip
            WHERE zip_start_dt >= DATE_SUB('2024-07-06', 29) AND zip_start_dt <= '2024-07-06'
            GROUP BY member_id, payment_method
        ) AS a
    ) AS b
    WHERE r = 1 -- 取第一就是最常用的支付方式
),
cart AS (
    SELECT
        member_id,
        COUNT(*) AS month1_cart_cnt_30, -- 加购次数
        SUM(sku_num) AS month1_cart_goods_cnt_30 -- 加购件数
    FROM dwd.dwd_yjx_cart_inc
    WHERE DATEDIFF('2024-07-06', dt) <= 30
    GROUP BY member_id
),
member AS (
    -- 有订单未必加购过，加购过未必提交了订单，所以 UNION 合并两种操作的用户，即可得到最全的用户信息
    -- 查询最近30日有过订单的用户
    SELECT member_id FROM dwd.dwd_yjx_order_zip
    WHERE zip_start_dt >= DATE_SUB('2024-07-06', 29) AND zip_start_dt <= '2024-07-06'
    UNION
    -- 查询最近30日有过加购的用户
    SELECT member_id FROM dwd.dwd_yjx_cart_inc
    WHERE DATEDIFF('2024-07-06', dt) <= 30
)
INSERT INTO TABLE ads.ads_profile_order_tag PARTITION (dt = '2024-07-06')

CREATE EXTERNAL TABLE IF NOT EXISTS ads.ads_user_active_cnt (
    calc_dt     STRING COMMENT '统计日期',
    day_count   BIGINT COMMENT '当日用户数量',
    wk_count    BIGINT COMMENT '当周用户数量',
    mn_count    BIGINT COMMENT '当月用户数量',
    is_weekend  STRING COMMENT '是否是周末Y,N，用于得到本周最终结果',
    is_monthend STRING COMMENT '是否是月末Y,N，用于得到本月最终结果'
)
COMMENT '活跃用户数表'
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','
LINES TERMINATED BY '\n'
LOCATION 'hdfs://hdfs-yjx/yjx/app/ads/ads_user_active_cnt';

-- 计算上一日的用户数量，并判断是否是周末和月末
WITH day_date AS (
    -- 计算上一日的活跃用户数
    SELECT guid
    FROM dws.dws_user_dau_bitmap_30d
    -- 参数 30，表示第 30 天也就是上一日。还有因为第一天是从 2^0 开始计算，所以第 30 天就是 2^29
    WHERE (CAST(bitmap AS BIGINT) & CAST(POW(2, 30 - 1) AS BIGINT)) > 0
),
day_result AS (
    SELECT
        '2024-07-30' AS dt,
        COUNT(DISTINCT guid) AS day_count,
        -- DAYOFWEEK：返回指定日期的星期，1 表示星期天，7 表示星期六
        IF(DAYOFWEEK('2024-07-30') IN (1), 'Y', 'N') AS is_weekend,
        IF(LAST_DAY('2024-07-30') = '2024-07-30', 'Y', 'N') AS is_monthend
    FROM day_date
),
-- 计算上一日所在自然周（周一到周日）的用户数量
week_active_dates AS (
    SELECT
        guid,
        DATE_ADD(NEXT_DAY('2024-07-30', 'MO') - 7, bit_pos) AS active_date
    FROM dws.dws_user_dau_bitmap_30d
    LATERAL VIEW POSEXPLODE(
        SPLIT(LPAD(BIN(bitmap), 30, '0'), '')
    ) t AS bit_pos, bit_val
    WHERE bit_val = '1'
    -- 周一：NEXT_DAY('2024-07-30', 'MO') - 7
    -- 周日：NEXT_DAY('2024-07-30', 'MO') - 1
    AND DATE_ADD(NEXT_DAY('2024-07-30', 'MO') - 7, bit_pos) BETWEEN (NEXT_DAY('2024-07-30', 'MO') - 7) AND (NEXT_DAY('2024-07-30', 'MO') - 1)
),
week_data AS (
    SELECT
        guid,
        COUNT(DISTINCT active_date) AS active_days
    FROM week_active_dates
    GROUP BY guid
),
week_result AS (
    SELECT
        '2024-07-30' AS dt,
        COUNT(DISTINCT guid) AS wk_count
    FROM week_data
),
-- 计算上一日所在自然月（月初到月末）的用户数量
month_active_dates AS (
    SELECT
        guid,
        DATE_ADD(TRUNC('2024-07-30', 'MM'), bit_pos) AS active_date
    FROM dws.dws_user_dau_bitmap_30d
    LATERAL VIEW POSEXPLODE(
        SPLIT(LPAD(BIN(bitmap), 30, '0'), '')
    ) t AS bit_pos, bit_val
    WHERE bit_val = '1'
    -- TRUNC：用于日期截断的函数，MM：表示截断到月的级别，最终结果为 2024-07-01
    -- LAST_DAY：返回指定日期所在月份最后一天，最终结果为 2024-07-31
    AND DATE_ADD(TRUNC('2024-07-30', 'MM'), bit_pos) BETWEEN TRUNC('2024-07-30', 'MM') AND LAST_DAY('2024-07-30')
),
month_data AS (
    SELECT
        '2024-07-30' AS dt,
        guid,
        COUNT(DISTINCT active_date) AS active_days
    FROM month_active_dates
    GROUP BY guid
),
month_result AS (
    SELECT
        '2024-07-30' AS dt,
        COUNT(DISTINCT guid) AS mn_count
    FROM month_data
)
-- 将三个子查询的结果合并，输出统计结果
SELECT
    d.dt AS calc_dt,
    d.day_count,
    w.wk_count,
    m.mn_count,
    d.is_weekend,
    d.is_monthend
FROM day_result d
JOIN week_result w
JOIN month_result m
ON d.dt = w.dt AND w.dt = m.dt;

    continuous_7days  INT COMMENT '本月内连续活跃 >= 7天的人数',
    continuous_15days INT COMMENT '本月内连续活跃 >= 15天的人数',
    continuous_20days INT COMMENT '本月内连续活跃 >= 20天的人数',

LINES TERMINATED BY '\n'
LOCATION 'hdfs://hdfs-yjx/yjx/app/ads/ads_user_active_cont_m';


WITH tmp AS (
    SELECT
        guid,
        -- 去掉左右空格，只处理有效的连续 1 段，统计连续 1 段的长度
        LENGTH(TRIM(segment)) AS continuous_days
    FROM dws.dws_user_dau_bitmap_30d
    -- 将二进制字符串按 0 分割，再将数组展平为多行，每一行表示一个连续 1 段
    LATERAL VIEW EXPLODE(SPLIT(BIN(bitmap), '0')) AS segment
    -- 过滤数组中的空值
    WHERE segment <> ''
),
result AS (
    SELECT guid, MAX(continuous_days) AS continuous_days
    FROM tmp
    GROUP BY guid
)
INSERT INTO TABLE ads.ads_user_active_cont_m
SELECT
    '2024-07-30' AS calc_dt,
    MONTH('2024-07-30') AS month,
    SUM(CASE WHEN continuous_days >= 5 THEN 1 ELSE 0 END) AS cnt_5_days,
    SUM(CASE WHEN continuous_days >= 7 THEN 1 ELSE 0 END) AS cnt_7_days,
    SUM(CASE WHEN continuous_days >= 15 THEN 1 ELSE 0 END) AS cnt_15_days,
    SUM(CASE WHEN continuous_days >= 20 THEN 1 ELSE 0 END) AS cnt_20_days,
    SUM(CASE WHEN continuous_days = 30 THEN 1 ELSE 0 END) AS cnt_30_days
FROM result;

-- 根据日期每天增量拉取计算
-- 计算上一日的新增用户数量，并判断是否是周末和月末
WITH day_data AS (
    SELECT
        '2024-07-06' AS dt,
        COUNT(DISTINCT guid) AS day_count,
        -- DAYOFWEEK：返回指定日期的星期，1 表示星期天，7 表示星期六
        IF(DAYOFWEEK(first_acc_dt) IN (1), 'Y', 'N') AS is_weekend,
        IF(LAST_DAY(first_acc_dt) = first_acc_dt, 'Y', 'N') AS is_monthend
    FROM dwd.dwd_user_register
    WHERE first_acc_dt = '2024-07-06'
    GROUP BY first_acc_dt
),
-- 计算上一日所在周的起始和结束日期，假设周一为第一天
week_boundary AS (
    SELECT
        NEXT_DAY('2024-07-06', 'MO') - 7 AS week_start, -- 2024-07-29
        NEXT_DAY('2024-07-06', 'MO') - 1 AS week_end -- 2024-08-04
),
-- 计算上一日所在自然周（周一到周日）的新增用户数量
week_data AS (
    SELECT
        '2024-07-06' AS dt,
        COUNT(DISTINCT guid) AS wk_count
    FROM dwd.dwd_user_register
    JOIN week_boundary wb ON first_acc_dt BETWEEN wb.week_start AND wb.week_end
),
-- 计算上一日所在自然月（月初到月末）的新增用户数量
month_data AS (
    SELECT
        '2024-07-06' AS dt,
        COUNT(DISTINCT guid) AS mn_count
    FROM dwd.dwd_user_register
    -- TRUNC：用于日期截断的函数，MM：表示截断到月的级别，最终结果为 2024-07-01
    -- LAST_DAY：返回指定日期所在月份最后一天，最终结果为 2024-07-31
    WHERE first_acc_dt BETWEEN TRUNC('2024-07-06', 'MM') AND LAST_DAY('2024-07-06')
)
-- 将三个子查询的结果合并，输出统计结果

LOCATION 'hdfs://hdfs-yjx/yjx/app/ads/ads_user_reactivation_cnt';

-- 超过 7 日未活跃的流失用户
WITH user_churn AS (
    SELECT
         '2024-07-08' AS calc_dt,
         guid AS guid
    FROM (
        SELECT
            guid,
            -- 最后一次未活跃，别担心 9999-12-31，因为 WHERE 过滤条件是 >= 7
            MAX(zip_end_dt) AS last_active_dt
        FROM dws.dws_user_active_range_zip
        GROUP BY guid
    ) t1
    -- 2024-07-08 减去 2024-07-01 等于 7，表示闭链为 2024-07-01，也就是说最后一次登录为 2024-06-30
    -- 8 号发现最后一次登录时间为 2024-06-30 的用户已经超过 7 日（含 7 日）未登录
    WHERE DATEDIFF('2024-07-08', last_active_dt) >= 7
),
-- 超过 7 日未活跃的用户，本日活跃了，且不是新增用户的数量
user_reactivation AS (
    SELECT guid
    FROM dws.dws_user_active_range_zip
    -- 新活跃且首访日期小于闭链日期 2024-07-09 表示老用户
    WHERE zip_end_dt = '9999-12-31' AND first_acc_dt < DATE_SUB('2024-07-08', 7)
    GROUP BY guid
)
SELECT
    '2024-07-16' AS calc_dt,
    COUNT(DISTINCT ur.guid) AS reactivation_cnt
FROM user_reactivation AS ur
LEFT JOIN user_churn AS uc ON ur.guid = uc.guid;

CREATE EXTERNAL TABLE IF NOT EXISTS ads.ads_user_retention_rate (
    calc_dt        STRING COMMENT '统计日期',
    first_acc_dt   STRING  COMMENT '新增用户日期',
    new_user_cnt   BIGINT COMMENT '新增用户数量',
    ret_days       INT COMMENT '截止当前日期留存天数',
    ret_user_cnt   BIGINT COMMENT  '留存用户数量',
    retention_rate DECIMAL(10,2) COMMENT '留存率'
)
COMMENT '新用户留存率表'
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','
LINES TERMINATED BY '\n'
LOCATION 'hdfs://hdfs-yjx/yjx/app/ads/ads_user_retention_rate';

-- 计算最近 30 日新增用户数量
WITH user_register AS (
    SELECT
        first_acc_dt,
        COUNT(DISTINCT guid) AS new_user_cnt
    FROM dwd.dwd_user_register
    -- 首访日期大于等于统计日期的 30 天前，小于等于统计日期
    WHERE first_acc_dt >= DATE_SUB('2024-07-07', 30) AND first_acc_dt <= '2024-07-07'
    GROUP BY first_acc_dt
),
-- 根据统计日期获取留存用户数量
user_retention AS (
    SELECT
        calc_dt,
        first_acc_dt,
        ret_days,
        ret_user_cnt
    FROM dws.dws_user_retention_cnt
    WHERE calc_dt = '2024-07-07'
)
-- 计算留存率，留存用户 ÷ 起始日新增用户数 × 100%
INSERT INTO TABLE ads.ads_user_retention_rate
SELECT
    a.calc_dt, -- 统计日期
    a.first_acc_dt, -- 新增用户日期
    b.new_user_cnt, -- 新增用户数量
    a.ret_days, -- 截止当前日期留存天数
    a.ret_user_cnt, -- 留存用户数量
    a.ret_user_cnt / b.new_user_cnt * 100 AS retention_rate -- 留存率
FROM user_retention a
LEFT JOIN user_register b ON a.first_acc_dt = b.first_acc_dt;
